#!/bin/bash
# Param 1 ($1) -> the percentage of lines
# Param 2 ($2) -> the input file
if [ -z $1 ] || [ -z $2 ]
then
        echo "Missing parameter, try again"
else
        lines_in_file=`wc -l < $2`
        desired_lines=$(($lines_in_file * $1 / 100))
        counter=1
        while [ $counter -le $desired_lines ];
        do
                random=`echo $(($RANDOM % $lines_in_file))`
                sed -n "${random}p" $2
                counter=$((counter + 1))
        done < $2

fi
